CGameObj has parent = 0, child = 0, sibling = 0, shortName,  initial, moved, click, 
take {
	purge 0, self; 
	"\n\nI picked up the " + makeHot(name,0,self) + ". ";
	move self to player;
	moved = true;
	updateInventory();
}, 
drop {
	purge 0, self;
	"\n\nI dropped the " + makeHot(name,0,self) + ". ";
	move self to player.parent;
	updateInventory();
}, examine {
	return style("popHeader") + cap(name) + style("popBody") + "\n" + description();
}, array = [1,2,3];

updateInventory() {
	setWindow(invWindow);
	clearWindow;
	"Inventory:";
	for each item of player {
		print makeHot("\nA " + item.name,0,item); 
	}
	setWindow(mainWindow);
};



listChildren(parentObj) {
	count = 0; result = "";
	numChildren = children(parentObj); 
	for each x of parentObj { 
		result += " a " + makeHot(x.name,0,x); 
		count += 1;
		if (count < numChildren -2)
			result = result + ",";
		if (count == numChildren -1)
			result = result + " and";
	}
	return result;
};
